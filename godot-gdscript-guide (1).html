Bienvenida // El entorno Godot

Godot 4 es un motor de videojuegos open source con un lenguaje propio llamado GDScript, diseñado para ser legible, rápido y muy integrado con el editor. Su sintaxis es similar a Python pero con tipos opcionales y acceso directo al engine.

⚡
Sin licencias
Godot es 100% libre y gratuito. Sin regalías ni restricciones comerciales.
🐍
GDScript simple
Sintaxis parecida a Python. Tipado estático opcional para mayor performance.
🎮
2D nativo
Sistema 2D completamente separado del 3D, con TileMap, física y shaders propios.
🔗
Señales (Events)
Sistema de signals para comunicación entre nodos sin acoplamiento fuerte.

Árbol de nodos (Scene Tree)

Todo en Godot son nodos. Los nodos se organizan en escenas jerárquicas. Una escena puede instanciarse dentro de otra, como si fueran prefabs.

🌲 WorldNode2D
├─🧍PlayerCharacterBody2D
│ ├─✨SpriteAnimatedSprite2D
│ └─💥CollisionShapeCollisionShape2D
├─🗺️TileMapTileMapLayer
└─📷Camera2DCamera2D
💡 Tip: En Godot 4.3+ el nodo TileMap fue reemplazado por TileMapLayer. Cada capa es un nodo separado, lo que permite mejor control por capas.

Instalación // Setup del proyecto

Descarga e instalación

Godot 4 no requiere instalación. Es un ejecutable portátil. Descárgalo desde godotengine.org. Elige la versión estándar (no .NET salvo que necesites C#).

  • Descargar Godot 4.x desde godotengine.org
  • Crear carpeta de proyecto (sin espacios ni tildes en la ruta)
  • Abrir Godot → New Project → seleccionar carpeta
  • Elegir "Mobile" o "Forward+" renderer (Forward+ para shaders avanzados)
  • Crear primera escena: Node2D como raíz
  • Guardar escena como main.tscn

    • Configuración para Pixel Art

    Antes de hacer cualquier sprite, configura el proyecto para pixel art:

    project.godot Project Settings → clave y valor 
    # Project Settings → Rendering → Textures
default_texture_filter = "Nearest"

# Project Settings → Display → Window
viewport_width  = 320   # resolución interna baja
viewport_height = 180
stretch_mode    = "canvas_items"   # escala pixel perfect
stretch_aspect  = "keep"
    ⚠️ Importante: Si no configuras Nearest como filtro de textura, los sprites tendrán antialiasing y se verán borrosos al escalar.

    Nodos y Escenas // Señales (Signals)

    Las signals son el sistema de eventos de Godot. Permiten que un nodo notifique a otros sin conocerlos directamente (bajo acoplamiento).

    GDScriptsignals.gd
    # ── Definir una signal propia ──
signal player_died
signal health_changed(new_health: int)
signal ability_unlocked(ability_name: String)

# ── Emitir la signal ──
func take_damage(amount: int) -> void:
    health -= amount
    health_changed.emit(health)   # notifica a todos
    if health <= 0:
        player_died.emit()

# ── Conectar en código ──
func _ready() -> void:
    # Conectar signal propia al HUD
    health_changed.connect($HUD.on_health_changed)

    # Señal de área (built-in)
    $HitBox.body_entered.connect(_on_hitbox_body_entered)

    # Con lambda
    player_died.connect(func(): _show_game_over())

# ── Receptor ──
func _on_hitbox_body_entered(body: Node2D) -> void:
    if body.is_in_group("enemies"):
        take_damage(10)
    💡 Patrón clave Metroidvania: Usa signals para que puertas, checkpoints y coleccionables notifiquen al GameManager sin referencias directas.

    Jugador 2D // CharacterBody2D

    CharacterBody2D es el nodo base para personajes controlados. Ofrece move_and_slide() que maneja colisiones automáticamente.

    🧍 PlayerCharacterBody2D
    ├─✨AnimatedSprite2DAnimatedSprite2D

    GDScriptPlayer.gd
    extends CharacterBody2D
class_name Player

# ── Parámetros ──
@export var speed: float      = 160.0
@export var jump_velocity: float = -340.0
@export var max_health: int   = 100

const GRAVITY = 980.0

var health: int
var is_dead: bool = false

signal health_changed(value: int)
signal died

@onready var anim: AnimatedSprite2D = $AnimatedSprite2D

func _ready() -> void:
    health = max_health

func _physics_process(delta: float) -> void:
    if is_dead: return

    # Aplicar gravedad
    if not is_on_floor():
        velocity.y += GRAVITY * delta

    # Saltar
    if Input.is_action_just_pressed("jump") and is_on_floor():
        velocity.y = jump_velocity

    # Movimiento horizontal
    var dir = Input.get_axis("move_left", "move_right")
    velocity.x = dir * speed

    # Voltear sprite
    if dir != 0:
        anim.flip_h = dir < 0

    move_and_slide()
    _update_animation()

func _update_animation() -> void:
    if not is_on_floor():
        anim.play("jump" if velocity.y < 0 else "fall")
    elif abs(velocity.x) > 10:
        anim.play("run")
    else:
        anim.play("idle")

func take_damage(amount: int) -> void:
    if is_dead: return
    health = max(0, health - amount)
    health_changed.emit(health)
    if health == 0: _die()

func _die() -> void:
    is_dead = true
    anim.play("die")
    died.emit()

# get_axis: devuelve -1, 0 o 1 suavizado
var dir = Input.get_axis("move_left", "move_right")

# is_action_just_pressed: solo el frame en que se pulsa
if Input.is_action_just_pressed("jump"):

if Input.is_action_pressed("attack"):

# ── Dash con cooldown ──
var can_dash: bool = true
var dash_cooldown: float = 0.8

func _physics_process(delta: float) -> void:
    if Input.is_action_just_pressed("dash") and can_dash:
        _do_dash()

func _do_dash() -> void:
    can_dash = false
    var dir = -1 if anim.flip_h else 1
    velocity.x = dir * 600
    # Resetear después del cooldown
    await get_tree().create_timer(dash_cooldown).timeout
    can_dash = true

# ── Coyote time y jump buffer (avanzado) ──
var coyote_time: float  = 0.12  # segundos tras caer del borde
var jump_buffer: float  = 0.1   # buffer de salto anticipado
var coyote_timer: float = 0.0
var buffer_timer: float = 0.0

func _handle_jump(delta: float) -> void:
    if is_on_floor(): coyote_timer = coyote_time
    else: coyote_timer -= delta

    if Input.is_action_just_pressed("jump"): buffer_timer = jump_buffer
    else: buffer_timer -= delta

    if buffer_timer > 0 and coyote_timer > 0:
        velocity.y = jump_velocity
        buffer_timer = 0
        coyote_timer = 0
    🕹️ Game Feel: Coyote time y jump buffer son las dos mecánicas más importantes para que el salto se sienta bien. ¡Implementarlas siempre!

    RoomManager: carga dinámica de salas

    GDScriptRoomManager.gd (Autoload)
    extends Node
class_name RoomManager

var current_room: Node = null
var player: Player

signal room_loaded(room: Node)

func load_room(path: String, spawn_point: String = "default") -> void:
    # 1. Fade out
    await Transition.fade_out()

    # 2. Descargar sala actual
    if current_room:
        current_room.queue_free()
        await get_tree().process_frame

    # 3. Cargar y añadir nueva sala
    var scene = load(path)
    current_room = scene.instantiate()
    get_tree().current_scene.add_child(current_room)

    # 4. Mover jugador al spawn point
    var sp = current_room.get_node_or_null("SpawnPoints/" + spawn_point)
    if sp and player:
        player.global_position = sp.global_position

    # 5. Fade in
    room_loaded.emit(current_room)
    await Transition.fade_in()

class_name EnemyBase

enum State { IDLE, PATROL, CHASE, ATTACK, HURT, DEAD }

@export var max_health: int    = 30
@export var move_speed: float  = 60.0
@export var detect_range: float = 150.0
@export var attack_range: float = 20.0
@export var damage: int         = 10

var state: State = State.PATROL
var health: int
var player: Player

const GRAVITY = 980.0

func _ready() -> void:
    health = max_health
    add_to_group("enemies")
    player = get_tree().get_first_node_in_group("player")

func _physics_process(delta: float) -> void:
    if not is_on_floor():
        velocity.y += GRAVITY * delta

    match state:
        State.PATROL: _patrol(delta)
        State.CHASE:  _chase(delta)
        State.ATTACK: _attack()
        _: pass

    move_and_slide()
    _update_state()

func _update_state() -> void:
    if not player or state == State.DEAD: return
    var dist = global_position.distance_to(player.global_position)
    if   dist <= attack_range: state = State.ATTACK
    elif dist <= detect_range: state = State.CHASE
    else:                       state = State.PATROL

func take_damage(amount: int) -> void:
    health -= amount
    if health <= 0: _die()

func _patrol(_d: float) -> void: pass  # implementar en subclase
func _chase(_d: float) -> void:
    var dir = sign(player.global_position.x - global_position.x)
    velocity.x = dir * move_speed
func _attack() -> void: pass
func _die() -> void:
    state = State.DEAD
    queue_free()

    Pixel Art // Shaders básicos
    GLSL / GDShaderoutline.gdshader (contorno sprite)
    shader_type canvas_item;

uniform vec4 outline_color: source_color = vec4(1.0, 1.0, 1.0, 1.0);
uniform float outline_width: hint_range(0.0, 5.0, 1.0) = 1.0;

void fragment() {
    vec4 col = texture(TEXTURE, UV);
    float alpha = col.a;

    // Samplear píxeles adyacentes
    float ox = outline_width * TEXTURE_PIXEL_SIZE.x;
    float oy = outline_width * TEXTURE_PIXEL_SIZE.y;
    float a = texture(TEXTURE, UV + vec2( ox,  0)).a
            + texture(TEXTURE, UV + vec2(-ox,  0)).a
            + texture(TEXTURE, UV + vec2( 0,  oy)).a
            + texture(TEXTURE, UV + vec2( 0, -oy)).a;

    // Si el pixel es transparente pero tiene vecino opaco = contorno
    if (alpha < 0.5 && a > 0.0) {
        COLOR = outline_color;
    } else {
        COLOR = col;
    }
}

const SAVE_PATH = "user://savegame.json"

func save() -> void:

func load_game() -> bool:
    if not FileAccess.file_exists(SAVE_PATH): return false
    var file  = FileAccess.open(SAVE_PATH, FileAccess.READ)
    var text  = file.get_as_text()
    file.close()
    var data = JSON.parse_string(text)
    if not data: return false

    # Restaurar estado
    var p = data["player_pos"]
    player.global_position = Vector2(p[0], p[1])
    player.health          = data["health"]
    AbilitySystem.abilities = data["abilities"]
    return true

enum GameState { PLAYING, PAUSED, MENU, CUTSCENE, GAME_OVER }

var state: GameState = GameState.PLAYING

signal state_changed(new_state: GameState)

func set_state(new_state: GameState) -> void:
    if state == new_state: return
    state = new_state
    state_changed.emit(new_state)

    set_state(GameState.PLAYING)
